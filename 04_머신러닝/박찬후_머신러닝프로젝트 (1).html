import numpy as np
def remove_outliers(data, multiplier=1.5):
    q1 = np.quantile(data, q=0.25)
    q3 = np.quantile(data, q=0.75) 
    iqr = q3 - q1
    lower_bound = q1 - iqr * multiplier
    upper_bound = q3 + iqr * multiplier
    
    return (data < lower_bound) | (data > upper_bound)
